1. создаём пакедж джсон yarn init, yarn add --dev typescript yarn add --dev ts-node
2. устанавливаем хардхэт yarn add --dev hardhat
3. создаём проект npx hardhat
3. устанавливаем необходимые плагины 
yarn add --dev @nomicfoundation/hardhat-toolbox @nomicfoundation/hardhat-network-helpers @nomicfoundation/hardhat-chai-matchers @nomiclabs/hardhat-ethers @nomiclabs/hardhat-etherscan chai ethers hardhat-gas-reporter solidity-coverage @typechain/hardhat typechain @typechain/ethers-v5 @ethersproject/abi @ethersproject/providers
4. yarn outdated - проверим нет ли новых версий
5. если есть yarn upgrade-interactive --latest
6. запуск тестов npx hardhat test
7. сгенерировать типы npx hardhat compile
8. посмотреть покрытие тестами npx hardhat coverage
9. добавляем линтер npm install -g solhint, чекнуть ошибки solhint 'contracts/**/*.sol'
10. устанавливаем енв yarn add --dev dotenv
11. deploy: npx hardhat run --network testnetBSC scripts/deploy-eventemitter.ts
12. verify: npx hardhat  verify --network testnetBSC 0xAe9f9b3032127D0C73F6BC0248D3A9aF9b64f292
